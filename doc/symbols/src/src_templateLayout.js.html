  6 (function (global) {
  7     var log = wef.logger("templateLayout"),
  8         templateLayout,

 13     /**
 14      * @class
 15      * TemplateLayout class
 16      * @param templateSource template definition
 17      */
 18     templateLayout = function (templateSource) {
 19         log.info("create templateLayout...");
 20         return new templateLayout.prototype.init(arguments);
 21     };

 23 
 24     templateLayout.prototype = {
 25         constructor:templateLayout,
 26         version:"0.0.1",
 27         templateSources:[],
 28         constants:{
 29             DISPLAY:"display",
 30             POSITION:"position"
 31         },
 32         compiler:null,
 33         generator:null,
 34 
 35         //templateLayout(), templateLayout(""), templateLayout("", "", ...)
 36         init:function (templateSources) {
 37             var args, firstSource, internalSources = [];
 38             log.debug("sources:", templateSources);
 39 
 40             log.debug("init subsystems...");
 41             parser = wef.cssParser();
 42             log.debug("subsystems... [OK]");
 43 
 44             args = Array.prototype.slice.call(templateSources);
 45             firstSource = args[0];
 46 
 47             //templateLayout()
 48             if (!firstSource) {
 49                 //TODO: load style & inline css
 50                 log.info("no external template loaded!!!");
 51 
 52                 Array.prototype.forEach.call(document.styleSheets, function (sheet) {
 53                     if (sheet.href !== null) {
 54                         //load external CSS
 55                         log.info("load external CSS", sheet.href);
 56                         internalSources.push(sheet.href);
 57                     }
 58                     else {
 59                         var text = sheet.ownerNode.innerHTML;
 60                         log.info("load style tag", text);
 61                         internalSources.push(text);
 62                     }
 63                 });
 64 
 65                 this.templateSources = internalSources.map(getContent);
 66                 log.info("templateLayout... [OK]");
 67                 return this;
 68             }
 69 
 70             //templateLayout("aString") and templateLayout("aString", "anotherString", ...)
 71             if (args.length >= 1 && args.every(isString)) {
 72                 this.templateSources = args.map(getContent);
 73                 log.info("templateLayout... [OK]");
 74                 return this;
 75             }
 76 
 77             log.error("Invalid argument");
 78             throw new Error("Invalid argument");
 79         },
 80         transform:function () {
 81             log.debug("transform...");
 82             var options = parseTransformOptions(arguments);
 83 
 84             if (options.parse) {
 85                 log.info("Step 1: parse");
 86                 log.group();
 87                 parser.whenStart(parserStarts);
 88                 parser.whenProperty(propertyFound);
 89                 parser.whenStop(parserDone);
 90 
 91                 parser.parse(this.templateSources.reduce(function (previous, source) {
 92                     return previous + source.sourceText;
 93                 }, ""));
 94 
 95                 log.groupEnd();
 96                 log.info("Step 1: parse... [OK]");
 97             }
 98             if (options.compile) {
 99                 log.info("Step 2: compile");
100                 log.group();
101                 tom = this.compiler().compile(buffer);
102                 // log.info("TOM: ", tom);
103                 log.groupEnd();
104                 log.info("Step 2: compile... [OK]");
105             }
106             if (options.generate) {
107                 log.info("Step 3: generate");
108                 log.group();
109                 this.generator(tom).patchDOM();
110                 log.groupEnd();
111                 log.info("Step 3: generate... [OK]");
112             }
113 
114             log.info("transform... [OK]");
115             return this;
116         },
117         //only for testing purposes
118         getBuffer:function () {
119             return buffer;
120         },
121         getTOM:function () {
122             return tom;
123         }
124     };
125 
126     templateLayout.fn = templateLayout.prototype;
127 
128     templateLayout.fn.init.prototype = templateLayout.fn;
129 
130     function isString(element) {
131         return typeof element == "string";
132     }
133 
134     function getSourceType(templateSource) {
135         var rxHttp = /^http[s]?:\/\/.*\.css$/i, rxFile = /^file:\/\/.*\.css$/i, rxPath = /^(?!\s*.*(http[s]?|file))(\.){0,2}(\/.*)*.*\.css$/i;
136         if (rxHttp.exec(templateSource)) {
137             return "http";
138         }
139         if (rxPath.exec(templateSource) || rxFile.exec(templateSource)) {
140             return "file";
141         }
142         return "css";
143     }
144 
145     function getContent(templateSource) {
146         var type = getSourceType(templateSource);
147         if (type == "http" || type == "file") {
148             return {
149                 type:type,
150                 sourceText:readFile(templateSource)
151             };
152         }
153         if (type == "css") {
154             return {
155                 type:type,
156                 sourceText:templateSource
157             };
158         } else {
159             throw new Error("unknown sourceType");
160         }
161     }
162 
163     function parseTransformOptions(args) {
164         var options = {parse:true, compile:true, generate:true};
165         if (args.length === 0) {
166             return options;
167         }
168         if (args[0].action == "none") {
169             options.parse = options.compile = options.generate = false;
170         }
171         if (args[0].action == "parse") {
172             options.compile = options.generate = false;
173         }
174         if (args[0].action == "compile") {
175             options.generate = false;
176         }
177         return options;
178     }
179 
180     function parserStarts(o) {
181         log.info("start parsing at", new Date(o.time).toLocaleTimeString());
182         buffer = {};
183     }
184 
185     function propertyFound(property) {
186         log.info("templateLayout listens: property found");
187         if (isSupportedProperty(property)) {
188             store(property);
189         }
190     }
191 
192     function parserDone(o) {
193         log.info("parsing done at", new Date(o.time).toLocaleTimeString());
194     }
195 
196     function readFile(url) {
197         //TODO: FIXME
198         function ajaxReadFile(url) {
199             var request = new XMLHttpRequest();
200             request.open("get", url, false);
201             request.send("");
202             log.info("request status: ", request.statusText);
203             return request.responseText;
204         }
205 
206         try {
207             log.info("reading file...");
208             var templateText = ajaxReadFile(url);
209             log.info("template loaded... [OK]");
210             return templateText;
211         } catch (e) {
212             log.error("Operation not supported", e);
213             throw new Error("Operation not supported", e);
214         }
215     }
216 
217     function store(rule) {
218         if (!buffer[rule.selectorText]) {
219             buffer[rule.selectorText] = {
220                 selectorText:rule.selectorText,
221                 declaration:{}
222             };
223         }
224         buffer[rule.selectorText].declaration[rule.declaration.property] = rule.declaration.valueText;
225         log.info("property stored: ", rule.declaration.property);
226     }
227 
228     function isSupportedProperty(rule) {
229         var iterator;
230         for (iterator in templateLayout.fn.constants) {
231             if (templateLayout.fn.constants.hasOwnProperty(iterator)) {
232                 if (templateLayout.fn.constants[iterator] == rule.declaration.property) {
233                     log.info("supported property found: ", rule.declaration.property);
234                     return true;
235                 }
236             }
237 
238         }
239         return false;
240     }
