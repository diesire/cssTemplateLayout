 33     function parseDisplay(displayValue) {
 34         /**
 35          * Name:    ‘display’
 36          * New value:    <display-type>? && [ [ <string> [ / <row-height> ]? ]+ ] <col-width>*
 37          * Percentages:    N/A
 38          * Computed value:    specified value
 39          *
 40          * The <display-type> is one of the following keywords. An omitted keyword is equivalent to ‘block’.
 41          * <display-type> = inline | block | list-item | inline-block | table | inline-table
 42          * | table-row-group | table-header-group | table-footer-group | table-row
 43          * | table-column-group | table-column | table-cell | table-caption | none
 44          *
 45          * Each <string> consist of one or more at signs (“@”), letters (or digits, see <letter> below),
 46          * periods (“.”) and spaces
 47          *
 48          * Each <row-height> sets the height of the preceding row. The default is ‘auto’.
 49          * The values can be as follows:
 50          * <length> An explicit height for that row. Negative values make the template illegal. If the length is
 51          * expressed in ‘gr’ units, these refer to the inherited grid, not the grid defined by the template itself.
 52          * auto The row's height is determined by its contents.
 53          * * (asterisk) All rows with an asterisk will be of equal height.
 54          *
 55          * Each <col-width> can be one of the following:
 56          * <length> An explicit width for that column. Negative values make the template illegal.
 57          * * (asterisk.) All columns with a ‘*’ have the same width. See the algorithm below.
 58          * max-content, min-content, minmax(p,q), fit-content
 59          */
 60         log.info("compiling display...");
 61         log.debug("display source: ", displayValue);
 62         var displayMetadata = {
 63             displayType:undefined,
 64             grid:[],
 65             widths:[]
 66         },
 67             allRefExp = /\s*(none|inline)?\s*(:?(\"[A-Za-z0-9\.@ ]+\")\s*(:?\/ ?(\*|\d+(:?px|%)))?)\s*((:?(:?(:?\d+(:?px|%))|\*)\s*)*)/gi,
 68             found,
 69             displayTypeFound,
 70             gridNotFound,
 71             completed;
 72         //all without displayType   (:?(\"[A-Za-z0-9\.@ ]+\")\s*(:?\/(\*|\d+(:?em|px|%)))?\s*)((:?(:?(:?\d+(:?em|px|%))|\*) *)*)
 73         // /\s*(inline|block|list-item|inline-block|table|inline-table|table-row-group|table-header-group|table-footer-group|table-row|table-column-group|table-column|table-cell|table-caption|none)?\s*(:?(\"[A-Za-z0-9\.@ ]+\")\s*(:?\/(\*|\d+(:?em|px|%)))?\s*)((:?(:?(:?\d+(:?em|px|%))|\*) *)*)/g
 74         if (displayValue !== undefined) {
 75             while ((found = allRefExp.exec(displayValue)) !== null) {
 76                 if (completed) {
 77                     log.error("Invalid template, invalid width definition");
 78                     throw new Error("Invalid template, width  definition");
 79                 }
 80                 if (found[1]) {
 81                     if (displayTypeFound) {
 82                         log.error("Invalid template, multiple display type");
 83                         throw new Error("Invalid template, multiple display type");
 84                     }
 85                     displayMetadata.displayType = found[1];
 86                     displayTypeFound = true;
 87                 }
 88                 if (found[3]) {
 89                     if (gridNotFound) {
 90                         log.error("Invalid template, invalid grid definition");
 91                         throw new Error("Invalid template, invalid grid definition");
 92                     }
 93                     displayMetadata.grid.push({rowText:found[3].replace(/"/g, "").replace(/\s*/, ""), height:undefined});
 94                 } else {
 95                     gridNotFound = true;
 96                 }
 97                 if (found[5]) {
 98                     if (!displayMetadata.grid[displayMetadata.grid.length - 1]) {
 99                         log.error("Invalid template, invalid height definition");
100                         throw new Error("Invalid template, height definition");
101                     }
102                     displayMetadata.grid[displayMetadata.grid.length - 1].height = found[5];
103                 }
104                 if (found[7]) {
105                     displayMetadata.widths = found[7].split(/\s+/);
106                     completed = true;
107                 }
108             }
109         }
110         log.info("display result: ", displayMetadata);
111         return displayMetadata;
112     }

114     function parsePosition(positionValue) {
115         var positionMetadata, matched, positionRegExp;
116         /**
117          * Name:    position
118          * New value:    <letter> | same
119          * Percentages:    N/A
120          * Computed value:    ‘<letter>’ or ‘static’; see text
121          *
122          * <letter> must be a single letter or digit, with category Lu, Ll, Lt or Nd in Unicode [UNICODE]),
123          * or a “@” symbol
124          */
125         log.info("compiling position...");
126         log.debug("position source: ", positionValue);
127         positionMetadata = {
128             position:null
129         };
130         positionRegExp = /^\s*(same|[a-zA-Z0-9])\s*$/i;
131         if (positionValue !== undefined) {
132             matched = positionValue.match(positionRegExp);
133             if (matched === null) {
134                 log.info("Unexpected value at ", positionValue);
135                 //throw new Error("Unexpected value at ", positionValue);
136                 return positionMetadata;
137             }
138             positionMetadata.position = matched[1];
139         }
140         log.info("position result: ", positionMetadata);
141         return positionMetadata;
142     }

144     function parseProperties(rule) {
145         var preProcessTemplate = {};
146         log.info("compiling properties...");
147         log.debug("properties source: ", rule);
148         preProcessTemplate.selectorText = rule.selectorText;
149         preProcessTemplate.display = parseDisplay(rule.declaration[templateLayout.fn.constants.DISPLAY]);
150         preProcessTemplate.position = parsePosition(rule.declaration[templateLayout.fn.constants.POSITION]);
151         log.info("properties result: ", preProcessTemplate);
152         return preProcessTemplate;
153     }

165     compiler.prototype = {
166         constructor:compiler,
167         init:function () {
168             return this;
169         },
170         compile:function (buffer) {
171             var selectorText, preProcessTemplate, inserted, template;
172             log.info("compile...");
173             log.debug("buffer: ", buffer);
174 
175             for (selectorText in buffer) {
176                 if (buffer.hasOwnProperty(selectorText)) {
177                     log.debug("next buffer element: ", selectorText);
178                     log.group();
179                     preProcessTemplate = parseProperties(buffer[selectorText]);
180                     if (this.isEmptyDisplay(preProcessTemplate.display) && this.isEmptyPosition(preProcessTemplate.position)) {
181                         log.groupEnd();
182                         log.info("preProcess: empty template", preProcessTemplate);
183                     } else {
184                         log.debug("preProcess:", preProcessTemplate);
185                         template = compiler.fn.templateBuilder(preProcessTemplate).createTemplate();
186                         inserted = rootTemplate.insert(template);
187                         log.groupEnd();
188                         log.info("element insertion...", inserted ? "[OK]" : "ERROR!");
189                     }
190                 }
191             }
192             log.debug("compile... OK");
193             return rootTemplate;
194         },
195         isEmptyDisplay:function (display) {
196             return display.grid.length === 0;
197         },
198         isEmptyPosition:function (position) {
199             return position.position === null;
200         }

304             rows:undefined,
305             filledSlots:undefined,
306             widths:[],
307             minWidths:[],
308             preferredWidths:[],
309             rowNumber:undefined,
310             colNumber:undefined,
311             init:function (rows, options) {
312                 this.rows = rows;
313                 this.filledSlots = {};
314                 //options
315                 this.widths = [];
316                 this.minWidths = [];
317                 this.preferredWidths = [];
318                 wef.extend(this, options, ["widths", "minWidths", "preferredWidths"]);
319                 this.colNumber = this.widths.length;
320                 this.rowNumber = rows.length;
321             },
322             hasSlot:function hasSlot(slotIdentifier) {
323                 var result;
324                 result = this.rows.some(function (row) {
325                     var regExp = new RegExp(slotIdentifier);
326                     return regExp.exec(row.rowText);
327                 });
328                 log.debug("hasSlot " + slotIdentifier + "?", result ? "yes" : "no");
329                 return result;
330             },
331             getDefaultSlot:function () {
332                 var firstLetterSlot, definedDefaultSlot = false;
333                 this.rows.forEach(function (row) {
334                     if (definedDefaultSlot) {
335                         return; //skip row
336                     }
337                     Array.prototype.some.call(row.rowText, function (slotText, slotIndex) {
338                         if (slotText === "@") {
339                             definedDefaultSlot = row.slots[slotIndex];
340                             return true;
341                         }
342                         if (!firstLetterSlot && slotText !== ".") {
343                             firstLetterSlot = row.slots[slotIndex];
344                             return false; //continue searching @
345                         }
346                     });
347                 });
348                 return definedDefaultSlot || firstLetterSlot;
349             },
350             setTemplate:function (aTemplate) {
351                 var row, tmp, result;
352                 if (this.hasSlot(aTemplate.position.position)) {
353                     //push template
354                     tmp = this.filledSlots[aTemplate.position.position] || [];
355                     tmp.push(aTemplate);
356                     this.filledSlots[aTemplate.position.position] = tmp;
357                     log.debug("grid [" + aTemplate.position.position + "] =", aTemplate);
358                     return true;
359                 } else {
360                     result = this.rows.some(function (row) {
361                         var result;
362                         result = row.slots.some(function (slotId) {
363                             var result;
364                             result = this.filledSlots[slotId.slotText] && this.filledSlots[slotId.slotText].some(function (currentTemplate) {
365                                 return !currentTemplate.isLeaf() && currentTemplate.insert(aTemplate);
366                             }, this);
367                             if (!result) {
368                                 log.debug("not found, try another slot");
369                             }
370                             return result;
371                         }, this);
372                         if (!result) {
373                             log.debug("not found, try another row");
374                         }
375                         return result;
376                     }, this);
377                     if (!result) {
378                         log.debug("not found, try another branch");
379                     }
380                     return result;
381                 }
382             }
383         };

443         function GridBuffer() {
444             this._rows = [];
445         }
446 
447         GridBuffer.prototype = {
448             constructor:GridBuffer,
449             _rows:[],
450             add:function (slot) {
451                 if (!Array.isArray(this._rows[slot.rowIndex])) {
452                     this._rows[slot.rowIndex] = [];
453                 }
454                 this._rows[slot.rowIndex][slot.colIndex] = slot;
455             },
456             getSlot:function (id) {
457                 var result = [];
458                 this._rows.forEach(function (row) {
459                     row.forEach(function (slot) {
460                         if (slot.slotText === id) {
461                             result.push(slot);
462                         }
463                     }, this);
464                 }, this);
465                 return result;
466             },
467             getRows:function () {
468                 return this._rows;
469             },
470             getRow:function (index) {
471                 return this._rows[index];
472             }
473         };

475         templateBuilder = function (source) {
476             log.debug("templateBuilder...");
477             return new templateBuilder.prototype.init(source);
478         };
479 
480         templateBuilder.prototype = {
481             constructor:templateBuilder,
482             buffer:undefined,
483             source:undefined,
484             init:function (source) {
485                 this.source = source;
486                 this.buffer = new GridBuffer();
487             },
488             createTemplate:function () {
489                 var display, grid, gridRows, gridOptions = {}, heights;
490                 display = this.source.display;
491                 grid = null;
492                 if (display.grid.length > 0) {
493                     this._addGrid(this.source.display);
494 
495                     heights = this._normalizeHeights(this.source.display);
496                     gridRows = this.buffer.getRows().map(function (row, index) {
497                         return compiler.fn.gridRow(this.source.display.grid[index].rowText, index, row, {height:heights[index]});
498                     }, this);
499 
500                     gridOptions.widths = this._normalizeWidths(this.source.display.widths);
501                     gridOptions.minWidths = gridOptions.widths.map(function (col) {
502                         return templateBuilder.fn._getMinWidth(col);
503                     });
504                     gridOptions.preferredWidths = gridOptions.widths.map(function (col) {
505                         return templateBuilder.fn._getPreferredWidth(col);
506                     });
507                     grid = compiler.fn.grid(gridRows, gridOptions);
508                 } else {
509                     //TODO:fixme
510                 }
511 
512                 return compiler.fn.template(this.source.selectorText, this.source.position, display, grid);
513             },
514             _addGrid:function (display) {
515                 if (display.grid.length > 0) {
516                     display.grid.forEach(function (row, rowIndex) {
517                         this._addGridRow(row, rowIndex);
518                     }, this);
519                 }
520             },
521             checkRowSpan:function (slotText, rowIndex, colIndex) {
522                 var previousRow, oldRowSpan, candidateRowSpan, slotGroups;
523                 slotGroups = this.buffer.getSlot(slotText);
524                 if (!slotGroups[0].allowRowSpan) {
525                     log.info("Slot don't allow row span");
526                     return false;
527                 }
528 
529                 if (rowIndex > 0) {
530                     return slotGroups.some(function (slot) {
531                         previousRow = this.source.display.grid[rowIndex - 1];
532                         if (previousRow.rowText.charAt(colIndex) === slotText) {
533                             oldRowSpan = slot.rowSpan;
534                             candidateRowSpan = (rowIndex - slot.rowIndex) + 1;
535                             if (candidateRowSpan == oldRowSpan) {
536                                 //do nothing
537                                 log.debug("Slot row span... [OK]");
538                                 return true;
539                             }
540                             if (candidateRowSpan == oldRowSpan + 1) {
541                                 slot.rowSpan++;
542                                 this.buffer.add(slot);
543                                 log.debug("Slot row span... [OK]");
544                                 return true;
545                             } else {
546                                 return false;
547                             }
548                         } else {
549                             return false;
550                         }
551                     }, this);
552                 } else {
553                     return false;
554                 }
555             },
556             checkColSpan:function (slotText, row, colIndex) {
557                 var slotGroups, oldColSpan, candidateColSpan;
558                 slotGroups = this.buffer.getSlot(slotText);
559                 if (!slotGroups[0].allowColSpan) {
560                     log.info("Slot don't allow col span");
561                     return false;
562                 }
563                 if (colIndex > 0) {
564                     return slotGroups.some(function (slot) {
565                         if (row.rowText[colIndex - 1] === slotText) {
566                             oldColSpan = slot.colSpan;
567                             candidateColSpan = (colIndex - slot.colIndex) + 1;
568                             if (candidateColSpan == oldColSpan + 1) {
569                                 slot.colSpan++;
570                                 this.buffer.add(slot);
571                                 log.debug("Slot col span... [OK]");
572                                 return true;
573                             } else {
574                                 return false;
575                             }
576                         } else {
577                             return false;
578                         }
579                     }, this);
580                 } else {
581                     return false;
582                 }
583             },
584             _addGridRow:function (row, rowIndex) {
585                 var identifiers, colSpan, rowSpan, regions;
586                 identifiers = Array.prototype.map.call(row.rowText, function (substring) {
587                     return substring.charAt(0);
588                 });
589 
590                 identifiers.forEach(function (slotText, colIndex) {
591                     //slot doesn't exist
592                     if (this.buffer.getSlot(slotText).length === 0) {
593                         this._addGridSlot(slotText, rowIndex, colIndex, 1, 1);
594                         return; //no span or region violations
595                     }
596 
597                     if (this.buffer.getSlot(slotText)[0].allowColSpan) {
598                         colSpan = this.checkColSpan(slotText, row, colIndex);
599                     } else {
600                         colSpan = false;
601                     }
602 
603                     if (this.buffer.getSlot(slotText)[0].allowRowSpan) {
604                         rowSpan = this.checkRowSpan(slotText, rowIndex, colIndex);
605                     } else {
606                         rowSpan = false;
607                     }
608 
609                     if (this.buffer.getSlot(slotText)[0].allowDisconnected) {
610                         //TODO: region check
611                         log.info("Slot allow disconnected regions");
612                         this._addGridSlot(slotText, rowIndex, colIndex, 1, 1);
613                         regions = true;
614                     } else {
615                         regions = false;
616                     }
617 
618                     if (!colSpan && !rowSpan && !regions) {
619                         throw new Error("Invalid template definition at \"" + slotText + "\"");
620                     }
621 
622                 }, this);
623             },
624             _addGridSlot:function (slotText, rowIndex, colIndex, rowSpan, colSpan) {
625                 var options, allowDisconnected, allowColSpan, allowRowSpan;
626 
627                 if (slotText === ".") {
628                     allowDisconnected = true;
629                     allowColSpan = false;
630                     allowRowSpan = false;
631                 } else {
632                     allowDisconnected = false;
633                     allowColSpan = true;
634                     allowRowSpan = true;
635                 }
636                 options = {
637                     rowSpan:rowSpan,
638                     colSpan:colSpan,
639                     allowDisconnected:allowDisconnected,
640                     allowColSpan:allowColSpan,
641                     allowRowSpan:allowRowSpan
642                 };
643                 this.buffer.add(compiler.fn.gridSlot(slotText, rowIndex, colIndex, options));
644             },
645             _getMinWidth:function (width) {
646                 if (width === "*") {
647                     return "0px";
648                 } else {
649                     return width;
650                 }
651             },
652             _getPreferredWidth:function (width) {
653                 if (width === "*") {
654                     return "9999px";
655                 } else {
656                     return width;
657                 }
658             },
659             _normalizeWidths:function (widths) {
660                 var i, tmp = [], dotColumn;
661 
662                 function checkDotColumn(row) {
663                     return row.rowText.charAt(i) === ".";
664                 }
665 
666                 for (i = 0; i < this._getMaxColNumber(this.source.display.grid); i++) {
667                     if (widths[i] === undefined) {
668                         tmp[i] = "*";
669                     } else {
670                         if (/-/.test(widths[i])) {
671                             throw new Error("Invalid argument: negative width not allowed");
672                         }
673                     }
674                     if (this.source.display.grid.every(checkDotColumn)) {
675                         tmp[i] = "0px";
676                     }
677                     if (widths[i] !== undefined) {
678                         tmp[i] = widths[i];
679                     }
680                 }
681                 return tmp;
682             },
683             _normalizeHeights:function (display) {
684                 var dotLineRegExp = /^\.+$/, negativeHeight = /-/, i, tmp = [];
685                 for (i = 0; i < display.grid.length; i++) {
686                     if (display.grid[i].height === undefined) {
687                         tmp[i] = "auto";
688                     } else {
689                         if (negativeHeight.test(display.grid[i].rowText)) {
690                             throw new Error("Invalid argument: negative height not allowed");
691                         }
692                     }
693                     if (dotLineRegExp.test(display.grid[i].rowText)) {
694                         tmp[i] = "0px";
695                     }
696                     //can overwrite dotLine rule
697                     if (display.grid[i].height !== undefined) {
698                         tmp[i] = display.grid[i].height;
699                     }
700                 }
701                 return tmp;
702             },
703             _getMaxColNumber:function (grid) {
704                 if (grid.length === 0) {
705                     return 0;
706                 }
707                 return grid.reduce(function (last, row) {
708                     return last > row.rowText.length ? last : row.rowText.length;
709                 }, 0);
710             }
711         };
