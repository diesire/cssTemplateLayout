  6 (function (wef) {
  7     var net;
  8 
  9     /**
 10      * @namespace
 11      */
 12     net = function () {
 13         return new net.prototype.init();
 14     };
 15 
 16     net.prototype = {
 17         constructor:net,
 18         /**
 19          * Version number
 20          */
 21         version:"0.1.0",
 22         /**
 23          * @ignore
 24          */
 25         init:function () {
 26             return this;
 27         },
 28         /**
 29          * Launch an XMLHttpRequest, waiting the result
 30          * @param url request url
 31          * @param [options] additional arguments
 32          * @param {string}options.method request method, supports[get|post]
 33          * @param {boolean}options.asynchronous request type, synchronous or asynchronous
 34          * @param {string}options.postBody message, in post request
 35          * @param {Function}options.success sucess callback
 36          * @param {Function}options.failure
 37          */
 38         ajax:function (url, options) {
 39             var request;
 40 
 41             function isSuccessfulRequest(request) {
 42                 return (request.status >= 200 && request.status < 300) || request.status == 304 || (request.status == 0 && request.responseText);
 43             }
 44 
 45             function respondToReadyState() {
 46                 if (request.readyState == 4) {
 47                     if (isSuccessfulRequest(request)) {
 48                         if (options.success) {
 49                             options.success(request);
 50                         }
 51                     } else {
 52                         if (options.failure) {
 53                             options.failure(request);
 54                         }
 55                     }
 56                 }
 57             }
 58 
 59             function setHeaders() {
 60                 var name, headers = {
 61                     "Accept":"text/javascript, text/html, application/xml, text/xml, */*"
 62                 };
 63                 for (name in headers) {
 64                     request.setRequestHeader(name, headers[name]);
 65                 }
 66             }
 67 
 68             request = xhr();
 69             if (typeof options === "undefined") {
 70                 options = {};
 71             }
 72 
 73             //TODO: refactor using wef.fn.extend
 74             options.method = options.method ? options.method.toLowerCase() : "get";
 75             options.asynchronous = options.asynchronous || true;
 76             options.postBody = options.postBody || "";
 77 
 78             request.onreadystatechange = respondToReadyState;
 79             request.open(options.method, url, options.asynchronous);
 80             setHeaders();
 81             request.send(options.postBody);
 82         }
 83     };
 84 
 85     function xhr() {
 86         if (typeof XMLHttpRequest !== "undefined" && (window.location.protocol !== "file:" || !window.ActiveXObject)) {
 87             return new XMLHttpRequest();
 88         } else {
 89             try {
 90                 return new ActiveXObject("Msxml2.XMLHTTP.6.0");
 91             } catch (e) {
 92             }
 93             try {
 94                 return new ActiveXObject("Msxml2.XMLHTTP.3.0");
 95             } catch (e) {
 96             }
 97             try {
 98                 return new ActiveXObject("Msxml2.XMLHTTP");
 99             } catch (e) {
100             }
101         }
102         return false;
103     }
104 
105     /**
106      * Extension point
107      */
108     net.fn = net.prototype;
109 
110     net.prototype.init.prototype = net.prototype;
111 
112     wef.net = net();
113 
114 })(window.wef);
